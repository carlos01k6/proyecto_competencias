SISTEMA DE EVALUACIÓN POR COMPETENCIAS AUTOMATIZADO
======================================================

ALGORITMO COMPLETO DEL PROYECTO

1. MÓDULO DE AUTENTICACIÓN
============================

INICIO_AUTENTICACION
  MOSTRAR pantalla de login
  
  REPETIR
    SOLICITAR email y contraseña
    
    SI email está vacío O contraseña está vacía ENTONCES
      MOSTRAR error "Campos requeridos"
      CONTINUAR
    FIN SI
    
    VALIDAR credenciales con Supabase Auth
    
    SI credenciales válidas ENTONCES
      OBTENER token JWT de Supabase
      OBTENER datos del usuario
      GUARDAR token en localStorage
      GUARDAR datos usuario en localStorage
      IR A dashboard
      SALIR del loop
    SINO
      MOSTRAR error "Email o contraseña incorrectos"
      LIMPIAR campos
    FIN SI
  HASTA QUE autenticación exitosa
  
FIN_AUTENTICACION


INICIO_REGISTRO
  MOSTRAR pantalla de registro
  
  REPETIR
    SOLICITAR: nombre, email, contraseña, rol
    
    SI nombre vacío O email vacío O contraseña vacío ENTONCES
      MOSTRAR error "Todos los campos son requeridos"
      CONTINUAR
    FIN SI
    
    SI email no es válido ENTONCES
      MOSTRAR error "Email inválido"
      CONTINUAR
    FIN SI
    
    CREAR usuario en Supabase Auth
    
    SI creación exitosa ENTONCES
      MOSTRAR mensaje "Cuenta creada exitosamente"
      REDIRIGIR a login
      SALIR del loop
    SINO
      MOSTRAR error de Supabase
    FIN SI
  HASTA QUE registro exitoso
  
FIN_REGISTRO


2. MÓDULO DE COMPETENCIAS
===========================

INICIO_LISTAR_COMPETENCIAS
  OBTENER usuario_id del token
  
  SI usuario_id es nulo ENTONCES
    MOSTRAR error "No autorizado"
    RETORNAR
  FIN SI
  
  SOLICITAR a Supabase: SELECT * FROM competencias WHERE docente_id = usuario_id
  
  SI hay competencias ENTONCES
    MOSTRAR lista de competencias en grid
    PARA CADA competencia EN lista
      MOSTRAR tarjeta con:
        - Nombre
        - Asignatura
        - Descripción
        - Descriptor
        - Botones: Ver Resultados, Editar, Eliminar
    FIN PARA
  SINO
    MOSTRAR mensaje "No hay competencias creadas"
  FIN SI
  
FIN_LISTAR_COMPETENCIAS


INICIO_CREAR_COMPETENCIA
  OBTENER usuario_id del token
  MOSTRAR formulario con campos:
    - Nombre (requerido)
    - Descripción
    - Descriptor (requerido)
    - Asignatura
  
  REPETIR
    SOLICITAR datos del formulario
    
    SI nombre vacío O descriptor vacío ENTONCES
      MOSTRAR error "Nombre y descriptor requeridos"
      CONTINUAR
    FIN SI
    
    GENERAR id_unico = UUID()
    CREAR registro EN Supabase:
      INSERT INTO competencias (id, nombre, descripcion, descriptor, asignatura, docente_id)
      VALUES (id_unico, nombre, descripcion, descriptor, asignatura, usuario_id)
    
    SI inserción exitosa ENTONCES
      MOSTRAR éxito "Competencia creada"
      RECARGAR lista de competencias
      CERRAR formulario
      SALIR del loop
    SINO
      MOSTRAR error "Error al crear"
    FIN SI
  HASTA QUE creación exitosa
  
FIN_CREAR_COMPETENCIA


INICIO_EDITAR_COMPETENCIA (competencia_id)
  OBTENER competencia actual de Supabase
  MOSTRAR formulario con datos de competencia
  
  REPETIR
    SOLICITAR cambios
    
    VALIDAR campos (nombre y descriptor requeridos)
    
    ACTUALIZAR EN Supabase:
      UPDATE competencias SET nombre, descripcion, descriptor, asignatura
      WHERE id = competencia_id
    
    SI actualización exitosa ENTONCES
      MOSTRAR éxito "Competencia actualizada"
      RECARGAR lista
      SALIR del loop
    SINO
      MOSTRAR error
    FIN SI
  HASTA QUE actualización exitosa
  
FIN_EDITAR_COMPETENCIA


INICIO_ELIMINAR_COMPETENCIA (competencia_id)
  SOLICITAR confirmación "¿Estás seguro?"
  
  SI usuario confirma ENTONCES
    ELIMINAR EN Supabase:
      DELETE FROM competencias WHERE id = competencia_id
    
    SI eliminación exitosa ENTONCES
      MOSTRAR éxito "Competencia eliminada"
      RECARGAR lista
    SINO
      MOSTRAR error
    FIN SI
  FIN SI
  
FIN_ELIMINAR_COMPETENCIA

4. MÓDULO DE CRITERIOS DE EVALUACIÓN
=====================================

INICIO_LISTAR_CRITERIOS (resultado_id)
  SOLICITAR a Supabase:
    SELECT * FROM criterios WHERE resultado_id = resultado_id
  
  CALCULAR total_ponderacion = SUMA(ponderacion) para todos los criterios
  
  SI hay criterios ENTONCES
    MOSTRAR lista de criterios
    MOSTRAR "Ponderación total: XX%"
    
    PARA CADA criterio EN lista
      MOSTRAR tarjeta con:
        - Nombre
        - Descripción
        - Ponderación
        - Indicador: "Requiere observación" (si aplica)
        - Botones: Editar, Eliminar
  SINO
    MOSTRAR "No hay criterios definidos"
  FIN SI
  
FIN_LISTAR_CRITERIOS


INICIO_CREAR_CRITERIO (resultado_id)
  MOSTRAR modal con formulario:
    - Nombre (requerido)
    - Descripción
    - Ponderación (0-100, defecto 100)
    - Checkbox: Requiere observación
  
  REPETIR
    SOLICITAR datos
    
    SI nombre vacío ENTONCES
      MOSTRAR error "Nombre requerido"
      CONTINUAR
    FIN SI
    
    SI ponderacion < 0 O ponderacion > 100 ENTONCES
      MOSTRAR error "Ponderación entre 0 y 100"
      CONTINUAR
    FIN SI
    
    GENERAR id_criterio = UUID()
    INSERTAR EN Supabase:
      INSERT INTO criterios (id, resultado_id, nombre, descripcion, ponderacion, requiere_observacion)
      VALUES (id_criterio, resultado_id, nombre, descripcion, ponderacion, requiere_observacion)
    
    SI inserción exitosa ENTONCES
      RECARGAR lista de criterios
      CERRAR modal
      SALIR del loop
    SINO
      MOSTRAR error
    FIN SI
  HASTA QUE creación exitosa
  
FIN_CREAR_CRITERIO


INICIO_EDITAR_CRITERIO (criterio_id)
  OBTENER criterio actual
  MOSTRAR modal con formulario
  
  REPETIR
    SOLICITAR cambios
    VALIDAR campos
    
    ACTUALIZAR EN Supabase:
      UPDATE criterios SET nombre, descripcion, ponderacion, requiere_observacion
      WHERE id = criterio_id
    
    SI actualización exitosa ENTONCES

      SALIR del loop

6. MÓDULO DE EVALUACIÓN (FUTURO)
=================================

INICIO_EVALUAR_ESTUDIANTE (estudiante_id, resultado_id)
  OBTENER criterios del resultado
  
  PARA CADA criterio EN criterios
    SOLICITAR calificación (0-100)
    
    SI criterio.requiere_observacion ENTONCES
      SOLICITAR observación/comentario
    FIN SI
    
    INSERTAR EN Supabase tabla "evaluaciones"
  FIN PARA
  
  CALCULAR calificacion_promedio = SUMA(calificacion * ponderacion) / 100
  
  INSERTAR EN Supabase tabla "calificaciones_finales"
  ACTUALIZAR seguimiento del estudiante
  
FIN_EVALUAR_ESTUDIANTE


7. MÓDULO DE REPORTES (FUTURO)
===============================

INICIO_GENERAR_REPORTE (tipo_reporte)

    OBTENER estudiante_id
    SOLICITAR competencia_id

      - Competencia

    OBTENER grupo/clase
    
    OBTENER DE Supabase:
      - Lista de estudiantes
      - Calificaciones de todos
      - Estadísticas generales
    
    GENERAR PDF con tabla comparativa
    DESCARGAR PDF
    
  FIN SI
  
FIN_GENERAR_REPORTE

9. VALIDACIONES GLOBALES
=========================

FUNCIÓN validar_token()
  OBTENER token de localStorage
  
  SI token no existe ENTONCES
    REDIRIGIR a login
    RETORNAR falso
  FIN SI
  
  SOLICITAR a Supabase validación de token
  
  SI token inválido O expirado ENTONCES
    LIMPIAR localStorage
    REDIRIGIR a login
    RETORNAR falso
  SINO
    RETORNAR verdadero
  FIN SI
  
FIN_FUNCIÓN


FUNCIÓN validar_permisos(usuario_rol)
  SI usuario_rol = "estudiante" ENTONCES
    PERMITIR solo lectura
    NO PERMITIR crear/editar competencias
  SINO SI usuario_rol = "docente" ENTONCES
    PERMITIR crear/editar competencias y criterios
    PERMITIR evaluar estudiantes
  SINO SI usuario_rol = "admin" ENTONCES
    PERMITIR todo
  FIN SI
  
FIN_FUNCIÓN


10. FLUJO PRINCIPAL DEL SISTEMA
================================

INICIO_APLICACION
  
  CARGAR aplicación
  
  VERIFICAR SI usuario autenticado
    SI no ENTONCES
      MOSTRAR página de login
    SINO
      VALIDAR token
      
      SI token válido ENTONCES
        CARGAR dashboard
        OBTENER datos del usuario
        MOSTRAR menú de navegación
        
        ESPERAR acción del usuario
        
        MIENTRAS usuario NO cierra sesión
          SI usuario selecciona "Competencias" ENTONCES
            EJECUTAR listar_competencias()
            ESPERAR acción: crear, editar, ver resultados, eliminar
            
            SI crear ENTONCES
              EJECUTAR crear_competencia()
            SINO SI editar ENTONCES
              EJECUTAR editar_competencia(id)
            SINO SI ver_resultados ENTONCES
              EJECUTAR listar_resultados(competencia_id)
              
              ESPERAR acción: crear, editar, ver criterios, eliminar
              
              SI crear_resultado ENTONCES
                EJECUTAR crear_resultado(competencia_id)
              SINO SI ver_criterios ENTONCES
                EJECUTAR listar_criterios(resultado_id)
                
                ESPERAR acción: crear, editar, eliminar criterios
                EJECUTAR según acción (crear_criterio, editar_criterio, eliminar_criterio)
              FIN SI
            FIN SI
            
          SINO SI usuario selecciona "Actividades" ENTONCES
            MOSTRAR página de actividades (en desarrollo)
            
          SINO SI usuario selecciona "Evaluación" ENTONCES
            MOSTRAR página de evaluación (en desarrollo)
            
          SINO SI usuario selecciona "Reportes" ENTONCES
            MOSTRAR página de reportes (en desarrollo)
            
          SINO SI usuario selecciona "Configuración" ENTONCES
            MOSTRAR página de configuración (en desarrollo)
            
          SINO SI usuario selecciona "Cerrar sesión" ENTONCES
            LIMPIAR localStorage
            REDIRIGIR a login
            SALIR del loop
            
          FIN SI
        FIN MIENTRAS
        
      SINO
        MOSTRAR error "Sesión expirada"
        REDIRIGIR a login
      FIN SI
    FIN SI
  
FIN_APLICACION


11. MANEJO DE ERRORES
=====================

PARA TODAS LAS OPERACIONES:
  
  INTENTAR
    EJECUTAR operación
  EXCEPTO error_red ENTONCES
    MOSTRAR "Error de conexión. Intenta de nuevo"
  EXCEPTO error_autenticacion ENTONCES
    LIMPIAR localStorage
    REDIRIGIR a login
  EXCEPTO error_validacion ENTONCES
    MOSTRAR mensaje de validación específico
  EXCEPTO error_base_datos ENTONCES
    MOSTRAR "Error en la base de datos"
    REGISTRAR error en logs
  EXCEPTO error_generico ENTONCES
    MOSTRAR "Ocurrió un error. Intenta de nuevo"
    REGISTRAR error en logs
  FIN INTENTAR


FIN DEL ALGORITMO COMPLETO
